{{- /* Shortcode: github-table Fetches a remote markdown file by URL and inlines a single section (extracted by heading name), rendered through the page's markdown renderer so headings, tables, and links render correctly. Renders correctly under both call forms ({{% github-table %}} and {{< github-table >}}) — the rebase pipeline rewrites percent-form shortcodes to angle-bracket form, so anything that relies on percent-form splice-back-into-source semantics breaks on rebased pages. Using RenderString here means both forms produce the same HTML. Parameters: - url: The remote URL of the markdown file to fetch. - section: The heading name of the section to extract. - exclude: (optional) A regex pattern for lines to remove from the section. */ -}} {{- $url := .Get "url" -}} {{ $exclude := .Get "exclude" }} {{- $section := .Get "section" -}} {{- with resources.GetRemote $url -}} {{- $content := .Content -}} {{- $pattern := printf "(?ms)^#+? %s\\s*(.*?)(^#+? |\\z)" $section -}} {{- $match := index (findRESubmatch $pattern $content) 0 -}} {{- if $match -}} {{- $section := index $match 1 | replaceRE (printf "^#+? %s\\s*" $section) "" -}} {{- if $exclude -}} {{- $section = replaceRE (printf "(?m)%s.*\n?" $exclude) "" $section -}} {{- end -}} {{- $.Page.RenderString $section -}} {{- else -}} {{- errorf "fail to find section: %s" $section -}} {{- end -}} {{- else -}} {{- errorf "unable to load content from %s" $url -}} {{- end -}}
