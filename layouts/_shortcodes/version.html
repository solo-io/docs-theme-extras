{{- /* Two supported config shapes for version metadata: 1. Flat (solo docs, kgateway, test fixture): Site.Params.versions = [...]; page URL is /{folder}/{linkVersion}/... 2. Per-section (agentgateway oss): Site.Params.sections.{name}.versions = [...]; page URL is /{folder}/{section}/{linkVersion}/... where section names are the keys of Site.Params.sections. Used when no flat list is set. */ -}} {{- $versions := .Site.Params.versions -}} {{- $condition := "" -}} {{- /* Dedent .Inner so block content (bullets, fenced code) at deep list-item indentation doesn't trip CommonMark's "4 spaces = code block" rule when RenderString processes the inner content in isolation. Without this, a version block at the 6-space continuation depth of a nested numbered list (the ambient-ecs pattern in solo-io/docs#2480) would render its bullets as a literal 
 instead of as a 
    . Computes the minimum
       leading-space count across non-empty lines and strips that many
       leading characters from every line. */ -}}
{{- $inner := .Inner -}}
{{- $lines := split $inner "\n" -}}
{{- $minIndent := -1 -}}
{{- range $lines -}}
  {{- $trimmed := strings.TrimLeft " \t" . -}}
  {{- if ne $trimmed "" -}}
    {{- $thisIndent := sub (len .) (len $trimmed) -}}
    {{- if or (eq $minIndent -1) (lt $thisIndent $minIndent) -}}
      {{- $minIndent = $thisIndent -}}
    {{- end -}}
  {{- end -}}
{{- end -}}
{{- if gt $minIndent 0 -}}
  {{- $dedented := slice -}}
  {{- range $lines -}}
    {{- if ge (len .) $minIndent -}}
      {{- $dedented = $dedented | append (substr . $minIndent) -}}
    {{- else -}}
      {{- $dedented = $dedented | append . -}}
    {{- end -}}
  {{- end -}}
  {{- $inner = delimit $dedented "\n" -}}
{{- end -}}

{{- /* Markdown-feature detection on $inner (post-dedent). If the inner
       content has no inline-formatting or list markers, RenderString
       would only damage it — turning `#` into 
    , "foo" into curly
       quotes, -- into –, plain text into HTML-escaped angle text
       inside a fenced code block. Emit raw in that case. Detects
       **bold**, lowercase HTML tags (terminated by space/`/`/`>` so
       yaml placeholders like  don't false-positive),
       backticks, [text](url) links, and bullet/numbered list markers
       at column 0-3. */ -}}
{{- $hasMarkdown := false -}}
{{- if findRE `\*\*` $inner 1 -}}{{- $hasMarkdown = true -}}{{- end -}}
{{- if not $hasMarkdown -}}
  {{- if findRE "`" $inner 1 -}}{{- $hasMarkdown = true -}}{{- end -}}
{{- end -}}
{{- if not $hasMarkdown -}}
  {{- if findRE `(?m)^[ \t]{0,3}([*+\-][ \t]|[0-9]+\.[ \t])` $inner 1 -}}{{- $hasMarkdown = true -}}{{- end -}}
{{- end -}}
{{- if not $hasMarkdown -}}
  {{- if findRE `\[[^\]]*\]\(` $inner 1 -}}{{- $hasMarkdown = true -}}{{- end -}}
{{- end -}}

{{- if $versions -}}
  {{- $version := .Page.FirstSection.RelPermalink -}}
  {{- $version_without_product := replace $version .Site.Params.folder "" -}}
  {{- $condition = trim $version_without_product "/" -}}
{{- else if .Site.Params.sections -}}
  {{- $versions = slice -}}
  {{- range $name, $data := .Site.Params.sections -}}
    {{- with $data.versions -}}
      {{- range . -}}
        {{- $versions = $versions | append . -}}
      {{- end -}}
    {{- end -}}
  {{- end -}}
  {{- $parts := split (trim .Page.RelPermalink "/") "/" -}}
  {{- range $i, $p := $parts -}}
    {{- if and (eq $condition "") (index $.Site.Params.sections $p) -}}
      {{- $next := add $i 1 -}}
      {{- if lt $next (len $parts) -}}
        {{- $condition = index $parts $next -}}
      {{- end -}}
    {{- end -}}
  {{- end -}}
{{- end -}}

{{- if ne $condition "" -}}
  {{- $include := .Get "include-if" | default "" -}}
  {{- $exclude := .Get "exclude-if" | default "" -}}
  {{- $include_if := split $include "," -}}
  {{- $exclude_if := split $exclude "," -}}
  {{- $currentProduct := .Site.Params.currentProduct | default "" -}}
  {{- $rendered := false -}}
  {{ range $versions }}
    {{- if in $condition .linkVersion -}}
       {{- $versionToml := .version -}}
       {{- $productMatches := or (not $currentProduct) (eq .product $currentProduct) -}}
           {{- if and (in $include_if $versionToml) (in $exclude_if $versionToml) -}}
              {{- errorf "Build condition %q appears in both include-if and exclude-if parameters" $condition -}}
           {{- end -}}
           {{- if isset $.Params "include-if" -}}
             {{- if and (in $include_if $versionToml) $productMatches (not $rendered) -}}
                {{- if not $hasMarkdown -}}
{{ $.Inner }}
                {{- else -}}
                  {{- $rendered_html := $.Page.RenderString (dict "display" "inline") $inner -}}
                  {{- if strings.Contains $rendered_html "[ \t]*\n[\\s]*<" "><" $rendered_html -}}
                    {{- replace $flat "\n" "
" | safeHTML -}}
                  {{- end -}}
                {{- end -}}
                {{- $rendered = true -}}
             {{- end -}}
             {{- else if isset $.Params "exclude-if" -}}
                {{- if and (not (in $exclude_if $versionToml)) $productMatches (not $rendered) -}}
                   {{- if not $hasMarkdown -}}
{{ $.Inner }}
                {{- else -}}
                  {{- $rendered_html := $.Page.RenderString (dict "display" "inline") $inner -}}
                  {{- if strings.Contains $rendered_html "[ \t]*\n[\\s]*<" "><" $rendered_html -}}
                    {{- replace $flat "\n" "
" | safeHTML -}}
                  {{- end -}}
                {{- end -}}
                   {{- $rendered = true -}}
             {{- end -}}
           {{- end -}}
    {{- end -}}
  {{- end -}}
{{- end -}}

   {{- if ne $include "" -}}
        {{- $versionImport := .Get "version" -}}

        {{- range $versions -}}

            {{- else if and .ossDir (eq .ossDir $versionImport) -}}
              {{- $versionAdjusted = .version -}}
            {{- end -}}

        {{- end -}}

        {{- $rendered := false -}}

               {{- if and (in $include $versionToml) (in $include $versionAdjusted) $productMatches (not $rendered) -}}
                      {{- if not $hasMarkdown -}}
{{ $.Inner }}

                    {{- replace $flat "\n" "

                  {{- end -}}
                {{- end -}}

               {{- end -}}
        {{- end -}}
  {{- end -}}
{{- end -}}
