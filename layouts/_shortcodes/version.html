{{- /* Local override of the version shortcode. Hugo doesn't expose whether a shortcode was called as {{% %}} or {{< >}} from inside the template, so this override dispatches on the SHAPE of .Inner to pick the right emit strategy: - TRAILING-STEP PATTERN (numbered or bullet list item at column 0-3 of .Inner): the version block continues a parent ordered or unordered list. The percent-form sidecar-manual-upgrade.md conref relies on this. Emit .Inner raw so Hugo re-flows the content through the surrounding markdown stream; that's the only way the parent list's structure stays continuous past the closer. Upstream's RenderString+flatten path produces a self-contained
    /
      ...
/
    HTML block that Goldmark treats as a block-HTML terminator, breaking the parent list (the "md leak" this override fixes). - NO-MARKDOWN PATTERN: .Inner has zero inline-formatting or list markers (no **, backticks, [text](url), HTML tags, or bullet/numbered list markers at column 0-3). Common for version blocks invoked INSIDE a fenced code block, where .Inner is yaml or shell text. RenderString would mangle this content — `#` becomes

    , "true" becomes “true”, -- becomes – — and that HTML gets escaped inside the fence, showing as literal angle-bracket text to the reader. Emit raw to keep the content as plain text. Detection is conservative: it avoids single-asterisk and underscore emphasis to prevent false positives in env-var names and shell glob patterns. - EVERYTHING ELSE (inline markdown like {{< version >}}**bold**, indented bullets/fences inside a deeper list-item continuation, angle-form block content): use the upstream dedent + RenderString + flatten approach. Required for any angle-bracket caller with markdown content because angle-bracket output bypasses markdown processing — raw .Inner would render bullets and fences as literal text. Regression guard: solo-io/docs#2480. The 
    -bypass lives in utils/flatten-rendered.html now:
         the partial early-returns when the rendered HTML contains
         …\n… chain breaks if the inter-span newlines
         get replaced with 
 entities — the parent's markdown
         parser then treats the inline `
    ` as inline HTML, applies
         CommonMark backslash-escape to `\<` inside the spans, and
         every "\" line-continuation marker renders as literal
         "" text. Keeping the original newlines lets the parent
         re-recognize the `
    ` as block HTML at column 0.

       Condition-resolution logic mirrors the upstream template. */ -}}

{{- $versions := .Site.Params.versions -}}
{{- $condition := "" -}}

{{- /* Shape detection (trailing-step, dedent, hasMarkdown, block, row) is
       centralized in utils/inner-shape so it stays in sync with conditional-text. */ -}}
{{- $shape := partial "utils/inner-shape.html" . -}}
{{- $isTrailingStep := $shape.isTrailingStep -}}
{{- $isFencedBlock := $shape.isFencedBlock -}}
{{- $inner := $shape.inner -}}
{{- $hasMarkdown := $shape.hasMarkdown -}}
{{- $isBlock := $shape.isBlock -}}
{{- $isRowContent := $shape.isRowContent -}}

{{- if $versions -}}
  {{- $version := .Page.FirstSection.RelPermalink -}}
  {{- $version_without_product := replace $version .Site.Params.folder "" -}}
  {{- $condition = trim $version_without_product "/" -}}

  {{- /* Path-segment fallback. The block above assumes URLs shaped like
         ///... (solo docs, test fixture). Consumers
         like kgw use /docs/
    //... where FirstSection
         is /docs/ and `folder` isn't set, so $condition lands on "docs"
         and matches nothing. When the derived $condition contains no
         known linkVersion, walk the URL segments and pick the first one
         that matches a linkVersion. Only fires when the original logic
         missed, so it can't break the solo / fixture paths. */ -}}
  {{- $matched := false -}}
  {{- range $versions -}}
    {{- if and (ne .linkVersion "") (in $condition .linkVersion) -}}
      {{- $matched = true -}}
    {{- end -}}
  {{- end -}}
  {{- if not $matched -}}
    {{- $parts := split (trim .Page.RelPermalink "/") "/" -}}
    {{- range $p := $parts -}}
      {{- range $versions -}}
        {{- if and (not $matched) (eq .linkVersion $p) -}}
          {{- $condition = $p -}}
          {{- $matched = true -}}
        {{- end -}}
      {{- end -}}
    {{- end -}}
  {{- end -}}
{{- else if .Site.Params.sections -}}
  {{- $versions = slice -}}
  {{- range $name, $data := .Site.Params.sections -}}
    {{- with $data.versions -}}
      {{- range . -}}
        {{- $versions = $versions | append . -}}
      {{- end -}}
    {{- end -}}
  {{- end -}}
  {{- $parts := split (trim .Page.RelPermalink "/") "/" -}}
  {{- range $i, $p := $parts -}}
    {{- if and (eq $condition "") (index $.Site.Params.sections $p) -}}
      {{- $next := add $i 1 -}}
      {{- if lt $next (len $parts) -}}
        {{- $condition = index $parts $next -}}
      {{- end -}}
    {{- end -}}
  {{- end -}}
{{- end -}}

{{- if ne $condition "" -}}
  {{- $include := .Get "include-if" | default "" -}}
  {{- $exclude := .Get "exclude-if" | default "" -}}
  {{- $include_if := split $include "," -}}
  {{- $exclude_if := split $exclude "," -}}
  {{- $currentProduct := .Site.Params.currentProduct | default "" -}}
  {{- $rendered := false -}}
  {{- range $versions -}}
    {{- if in $condition .linkVersion -}}
       {{- $versionToml := .version -}}
       {{- $productMatches := or (not $currentProduct) (eq .product $currentProduct) -}}
       {{- if and (in $include_if $versionToml) (in $exclude_if $versionToml) -}}
         {{- errorf "Build condition %q appears in both include-if and exclude-if parameters" $condition -}}
       {{- end -}}
       {{- if isset $.Params "include-if" -}}
         {{- if and (in $include_if $versionToml) $productMatches (not $rendered) -}}
           {{- partial "utils/emit-inner.html" (dict "page" $.Page "rawInner" $.Inner "inner" $inner "isTrailingStep" $isTrailingStep "isFencedBlock" $isFencedBlock "isRowContent" $isRowContent "hasMarkdown" $hasMarkdown "isBlock" $isBlock) -}}
           {{- $rendered = true -}}
         {{- end -}}
       {{- else if isset $.Params "exclude-if" -}}
         {{- if and (not (in $exclude_if $versionToml)) $productMatches (not $rendered) -}}
           {{- partial "utils/emit-inner.html" (dict "page" $.Page "rawInner" $.Inner "inner" $inner "isTrailingStep" $isTrailingStep "isFencedBlock" $isFencedBlock "isRowContent" $isRowContent "hasMarkdown" $hasMarkdown "isBlock" $isBlock) -}}
           {{- $rendered = true -}}
         {{- end -}}
       {{- end -}}
    {{- end -}}
  {{- end -}}
{{- end -}}

{{- if eq $condition "" -}}
  {{- $include := .Get "include-if" | default "" -}}
  {{- if ne $include "" -}}
    {{- $versionImport := .Get "version" -}}
    {{- $versionAdjusted := "" -}}
    {{- range $versions -}}
      {{- if not $versionAdjusted -}}
        {{- if or (eq .version $versionImport) (eq .linkVersion $versionImport) -}}
          {{- $versionAdjusted = .version -}}
        {{- else if and .ossDir (eq .ossDir $versionImport) -}}
          {{- $versionAdjusted = .version -}}
        {{- end -}}
      {{- end -}}
    {{- end -}}
    {{- $currentProduct := .Site.Params.currentProduct | default "" -}}
    {{- $rendered := false -}}
    {{- range $versions -}}
      {{- $versionToml := .version -}}
      {{- $productMatches := or (not $currentProduct) (eq .product $currentProduct) -}}
      {{- if and (in $include $versionToml) (in $include $versionAdjusted) $productMatches (not $rendered) -}}
        {{- partial "utils/emit-inner.html" (dict "page" $.Page "rawInner" $.Inner "inner" $inner "isTrailingStep" $isTrailingStep "isFencedBlock" $isFencedBlock "isRowContent" $isRowContent "hasMarkdown" $hasMarkdown "isBlock" $isBlock) -}}
        {{- $rendered = true -}}
      {{- end -}}
    {{- end -}}
  {{- end -}}
{{- end -}}

{{- if eq $condition "" -}}
  {{- $exclude := .Get "exclude-if" | default "" -}}
  {{- if ne $exclude "" -}}
    {{- $versionImport := .Get "version" -}}
    {{- $versionAdjusted := "" -}}
    {{- range $versions -}}
      {{- if not $versionAdjusted -}}
        {{- if or (eq .version $versionImport) (eq .linkVersion $versionImport) -}}
          {{- $versionAdjusted = .version -}}
        {{- else if and .ossDir (eq .ossDir $versionImport) -}}
          {{- $versionAdjusted = .version -}}
        {{- end -}}
      {{- end -}}
    {{- end -}}
    {{- $currentProduct := .Site.Params.currentProduct | default "" -}}
    {{- $rendered := false -}}
    {{- range $versions -}}
      {{- $versionToml := .version -}}
      {{- $productMatches := or (not $currentProduct) (eq .product $currentProduct) -}}
      {{- if and (not (or (in $exclude $versionToml) (in $exclude $versionAdjusted))) $productMatches (not $rendered) -}}
        {{- partial "utils/emit-inner.html" (dict "page" $.Page "rawInner" $.Inner "inner" $inner "isTrailingStep" $isTrailingStep "isFencedBlock" $isFencedBlock "isRowContent" $isRowContent "hasMarkdown" $hasMarkdown "isBlock" $isBlock) -}}
        {{- $rendered = true -}}
      {{- end -}}
    {{- end -}}
  {{- end -}}
{{- end -}}
