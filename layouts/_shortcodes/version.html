{{- /* Two supported config shapes for version metadata: 1. Flat (solo docs, kgateway, test fixture): Site.Params.versions = [...]; page URL is /{folder}/{linkVersion}/... 2. Per-section (agentgateway oss): Site.Params.sections.{name}.versions = [...]; page URL is /{folder}/{section}/{linkVersion}/... where section names are the keys of Site.Params.sections. Used when no flat list is set. */ -}} {{- $versions := .Site.Params.versions -}} {{- $condition := "" -}} {{- /* Dedent .Inner so block content (bullets, fenced code) at deep list-item indentation doesn't trip CommonMark's "4 spaces = code block" rule when RenderString processes the inner content in isolation. Without this, a version block at the 6-space continuation depth of a nested numbered list (the ambient-ecs pattern in solo-io/docs#2480) would render its bullets as a literal 
 instead of as a 
    . Computes the minimum
       leading-space count across non-empty lines and strips that many
       leading characters from every line. */ -}}
{{- $inner := .Inner -}}
{{- $lines := split $inner "\n" -}}
{{- $minIndent := -1 -}}
{{- range $lines -}}
  {{- $trimmed := strings.TrimLeft " \t" . -}}
  {{- if ne $trimmed "" -}}
    {{- $thisIndent := sub (len .) (len $trimmed) -}}
    {{- if or (eq $minIndent -1) (lt $thisIndent $minIndent) -}}
      {{- $minIndent = $thisIndent -}}
    {{- end -}}
  {{- end -}}
{{- end -}}
{{- if gt $minIndent 0 -}}
  {{- $dedented := slice -}}
  {{- range $lines -}}
    {{- if ge (len .) $minIndent -}}
      {{- $dedented = $dedented | append (substr . $minIndent) -}}
    {{- else -}}
      {{- $dedented = $dedented | append . -}}
    {{- end -}}
  {{- end -}}
  {{- $inner = delimit $dedented "\n" -}}
{{- end -}}

{{- /* Trailing-step detection. Fires when the FIRST non-blank line of
       .Inner is a list-item marker (column 0-3) AND the body contains
       no block-level markdown (no headings, no fenced code) that would
       break raw emission. In that case the version block is continuing
       a parent list, and the safe emission strategy is raw .Inner — the
       surrounding markdown stream then sees real list markers and
       merges them into the neighboring list. Used in tandem with the
       list-item conversion in reuse.html / rebase.html, which rewrites
       angle-form `{{< version >}}* item` wraps to percent form so this
       raw emit reaches Goldmark via the reparse pass instead of as
       opaque post-Goldmark substitution. RenderString-then-flatten
       would otherwise emit `
    • …
    `, and the resulting
       type-7 HTML block absorbs any subsequent bare bullets (`* item`)
       as literal text until the next blank line (the `jwt-about.md`
       "Istio docs for [JWT tokens]" leak, the `release-notes.md`
       Upgrade/Version leak). The fence/heading exclusion preserves the
       solo-io/docs#2480 nested-list+fence regression guard, which
       relies on RenderString-then-flatten so the inner `
      ` and
       `
      ` reach Goldmark as a single HTML placeholder substitution
       instead of as raw indented markdown that the parent 
        's
       lazy-continuation rules would otherwise mangle. */ -}}
{{- $isTrailingStep := false -}}
{{- $firstNonBlank := "" -}}
{{- range (split $inner "\n") -}}
  {{- if and (eq $firstNonBlank "") (ne (trim . " \t") "") -}}
    {{- $firstNonBlank = . -}}
  {{- end -}}
{{- end -}}
{{- $hasBlockMarkdown := false -}}
{{- if findRE "(?m)^[ \\t]{0,3}(#{1,6}\\s|`{3})" $inner 1 -}}
  {{- $hasBlockMarkdown = true -}}
{{- end -}}
{{- if and (not $hasBlockMarkdown) (findRE `^[ \t]{0,3}([*+\-][ \t]|[0-9]+\.[ \t])` $firstNonBlank 1) -}}
  {{- $isTrailingStep = true -}}
{{- end -}}

{{- /* Table-row content detection. Fires when every non-blank line of
       .Inner is a pipe-delimited table row (starts and ends with `|`).
       Without this, a percent-form version block whose inner is one or
       more table rows with inline backticks/links — $hasMarkdown=true —
       routes through RenderString(inline), which emits each row as a
       `
        …
        ` with literal pipes (the Solo docs hub istio
       `pipeline-metrics.md` leak, where a multi-row `{{% version %}}` in
       the middle of a backticked metrics table rendered as `
        |…
       |
        ` paragraphs that broke the parent ``). Raw
       emit lets the percent-form re-flow reinsert the rows verbatim, so
       Goldmark sees them as siblings of the surrounding table rows and
       parses them into the same ``. Restricted to inputs where
       every non-blank line is a row so prose with stray pipes or table
       bodies mixed with prose don't accidentally route here. */ -}}
{{- $isRowContent := false -}}
{{- $hasAnyRow := false -}}
{{- $allLinesAreRows := true -}}
{{- range (split $inner "\n") -}}
  {{- $t := trim . " \t" -}}
  {{- if ne $t "" -}}
    {{- if and (hasPrefix $t "|") (hasSuffix $t "|") -}}
      {{- $hasAnyRow = true -}}
    {{- else -}}
      {{- $allLinesAreRows = false -}}
    {{- end -}}
  {{- end -}}
{{- end -}}
{{- if and $hasAnyRow $allLinesAreRows -}}
  {{- $isRowContent = true -}}
{{- end -}}

{{- /* Markdown-feature detection on $inner (post-dedent). If the inner
       content has no inline-formatting or list markers, RenderString
       would only damage it — turning `#` into 
        , "foo" into curly
       quotes, -- into –, plain text into HTML-escaped angle text
       inside a fenced code block. Emit raw in that case. Detects
       **bold**, lowercase HTML tags (terminated by space/`/`/`>` so
       yaml placeholders like  don't false-positive),
       backticks, [text](url) links, and bullet/numbered list markers
       at column 0-3. */ -}}
{{- $hasMarkdown := false -}}
{{- if findRE `\*\*|<(/?[a-z][a-z0-9]*[\s/>])` $inner 1 -}}{{- $hasMarkdown = true -}}{{- end -}}
{{- if not $hasMarkdown -}}
  {{- if findRE "`" $inner 1 -}}{{- $hasMarkdown = true -}}{{- end -}}
{{- end -}}
{{- if not $hasMarkdown -}}
  {{- if findRE `(?m)^[ \t]{0,3}([*+\-][ \t]|[0-9]+\.[ \t])` $inner 1 -}}{{- $hasMarkdown = true -}}{{- end -}}
{{- end -}}
{{- if not $hasMarkdown -}}
  {{- if findRE `\[[^\]]*\]\(` $inner 1 -}}{{- $hasMarkdown = true -}}{{- end -}}
{{- end -}}

{{- if $versions -}}
  {{- $version := .Page.FirstSection.RelPermalink -}}
  {{- $version_without_product := replace $version .Site.Params.folder "" -}}
  {{- $condition = trim $version_without_product "/" -}}

  {{- /* Path-segment fallback. The block above assumes URLs shaped like
         ///... (solo docs, test fixture). Consumers
         like kgw use /docs/
        //... where FirstSection
         is /docs/ and `folder` isn't set, so $condition lands on "docs"
         and matches nothing. When the derived $condition contains no
         known linkVersion, walk the URL segments and pick the first one
         that matches a linkVersion. Only fires when the original logic
         missed, so it can't break the solo / fixture paths. */ -}}
  {{- $matched := false -}}
  {{- range $versions -}}
    {{- if and (ne .linkVersion "") (in $condition .linkVersion) -}}
      {{- $matched = true -}}
    {{- end -}}
  {{- end -}}
  {{- if not $matched -}}
    {{- $parts := split (trim .Page.RelPermalink "/") "/" -}}
    {{- range $p := $parts -}}
      {{- range $versions -}}
        {{- if and (not $matched) (eq .linkVersion $p) -}}
          {{- $condition = $p -}}
          {{- $matched = true -}}
        {{- end -}}
      {{- end -}}
    {{- end -}}
  {{- end -}}
{{- else if .Site.Params.sections -}}
  {{- $versions = slice -}}
  {{- range $name, $data := .Site.Params.sections -}}
    {{- with $data.versions -}}
      {{- range . -}}
        {{- $versions = $versions | append . -}}
      {{- end -}}
    {{- end -}}
  {{- end -}}
  {{- $parts := split (trim .Page.RelPermalink "/") "/" -}}
  {{- range $i, $p := $parts -}}
    {{- if and (eq $condition "") (index $.Site.Params.sections $p) -}}
      {{- $next := add $i 1 -}}
      {{- if lt $next (len $parts) -}}
        {{- $condition = index $parts $next -}}
      {{- end -}}
    {{- end -}}
  {{- end -}}
{{- end -}}

{{- if ne $condition "" -}}
  {{- $include := .Get "include-if" | default "" -}}
  {{- $exclude := .Get "exclude-if" | default "" -}}
  {{- $include_if := split $include "," -}}
  {{- $exclude_if := split $exclude "," -}}
  {{- $currentProduct := .Site.Params.currentProduct | default "" -}}
  {{- $rendered := false -}}
  {{ range $versions }}
    {{- if in $condition .linkVersion -}}
       {{- $versionToml := .version -}}
       {{- $productMatches := or (not $currentProduct) (eq .product $currentProduct) -}}
           {{- if and (in $include_if $versionToml) (in $exclude_if $versionToml) -}}
              {{- errorf "Build condition %q appears in both include-if and exclude-if parameters" $condition -}}
           {{- end -}}
           {{- if isset $.Params "include-if" -}}
             {{- if and (in $include_if $versionToml) $productMatches (not $rendered) -}}
                {{- if or $isTrailingStep $isRowContent (not $hasMarkdown) -}}
{{ $.Inner }}
                {{- else -}}
                  {{- $rendered_html := $.Page.RenderString (dict "display" "inline") $inner -}}
                  {{- if strings.Contains $rendered_html "[ \t]*\n[\\s]*<" "><" $rendered_html -}}
                    {{- replace $flat "\n" "
" | safeHTML -}}
                  {{- end -}}
                {{- end -}}
                {{- $rendered = true -}}
             {{- end -}}
             {{- else if isset $.Params "exclude-if" -}}
                {{- if and (not (in $exclude_if $versionToml)) $productMatches (not $rendered) -}}
                   {{- if or $isTrailingStep $isRowContent (not $hasMarkdown) -}}
{{ $.Inner }}
                {{- else -}}
                  {{- $rendered_html := $.Page.RenderString (dict "display" "inline") $inner -}}
                  {{- if strings.Contains $rendered_html "[ \t]*\n[\\s]*<" "><" $rendered_html -}}
                    {{- replace $flat "\n" "
" | safeHTML -}}
                  {{- end -}}
                {{- end -}}
                   {{- $rendered = true -}}
             {{- end -}}
           {{- end -}}
    {{- end -}}
  {{- end -}}
{{- end -}}

{{- if eq $condition "" -}}
   {{- $include := .Get "include-if" | default "" -}}
   {{- if ne $include "" -}}
        {{- $versionImport := .Get "version" -}}
        {{- $versionAdjusted := "" -}}
        {{- range $versions -}}
          {{- if not $versionAdjusted -}}
            {{- if or (eq .version $versionImport) (eq .linkVersion $versionImport) -}}

            {{- else if and .ossDir (eq .ossDir $versionImport) -}}
              {{- $versionAdjusted = .version -}}
            {{- end -}}
          {{- end -}}
        {{- end -}}
        {{- $currentProduct := .Site.Params.currentProduct | default "" -}}
        {{- $rendered := false -}}
        {{- range $versions -}}
            {{- $versionToml := .version -}}
            {{- $productMatches := or (not $currentProduct) (eq .product $currentProduct) -}}
               {{- if and (in $include $versionToml) (in $include $versionAdjusted) $productMatches (not $rendered) -}}
                      {{- if or $isTrailingStep $isRowContent (not $hasMarkdown) -}}
{{ $.Inner }}

                  {{- if strings.Contains $rendered_html "[ \t]*\n[\\s]*<" "><" $rendered_html -}}
