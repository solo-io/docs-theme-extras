{{- /* rebase: render content from another version's asset tree into the current page, with version-aware shortcode rewrites applied. Pipeline overview — each block below is one stage: 1. SETUP — resolve file path, page version, product, and look up the resource. Falls back to bare path if the versioned path isn't on disk. 2. CONTENT PREP — strip front matter; mark page for Mermaid / manual-cards detection. 3. FORM CONVERSION — bulk-rewrite percent-form shortcodes to angle- form so RenderString processes them, then revert specific shortcodes back to percent- form when their inner content must re-flow through markdown (tab, steps, include). 4. ARG INJECTION — pass the resolved version + product into shortcodes that need them (link-hextra, reuse). 5. VERSION REMAP — handle keepVersion= on the version shortcode, then two-pass OSS↔enterprise version-string swap via indexed placeholders. 6. RENDER — hand the transformed markdown to RenderString. When adding a new shortcode that needs special treatment under rebase, find the right stage and add the rule there. The structural- HTML parity test in versioning.spec.ts (everything vs rebased) is the safety net — if you forget a stage, that test will fail. */ -}} {{/* ─── Stage 1: SETUP ──────────────────────────────────────────────── */}} {{ $file := .Get "file" }} {{/* Set to default value, here agentgateway. This is required so that the product value will persist after the if statement. */}} {{ $product := "agentgateway" }} {{/* Resolve the content version from the page URL and prepend // to the file path so it resolves to the assembled directory (e.g. agw-docs/kubernetes/... → agentgateway/2.3.x/agw-docs/kubernetes/...). The first segment that matches a known version wins. */}} {{ $version := "" }} {{ $urlParts := split .Page.RelPermalink "/" }} {{ range $urlPart := $urlParts }} {{ if not $version }} {{ range $.Site.Params.versions }} {{ if eq .version $urlPart }} {{ $version = .version }} {{ $product = .product | default "agentgateway" }} {{ end }} {{ end }} {{ end }} {{ end }} {{ $currentProduct := $.Site.Params.currentProduct | default "" }} {{ $doc := false }} {{ if and $version $currentProduct }} {{ $versionedFile := printf "%s/%s/%s" $currentProduct $version $file }} {{ $doc = resources.Get $versionedFile }} {{ if $doc }} {{ $file = $versionedFile }} {{ end }} {{ end }} {{ if not $doc }} {{ $doc = resources.Get $file }} {{ end }} {{ if not $doc }} {{ errorf "rebase: resource not found: %q (page: %s). Run 'make server PRODUCT=' to assemble versioned assets." (.Get "file") .Page.RelPermalink }} {{ end }} {{ $content := $doc.Content }} {{/* For kgateway/envoy content, override product so link-hextra can apply envoy-specific path transformations (e.g. reference/api → /reference/api/kgateway). Detected before shortcode conversion, so it catches both {{< cards >}} and {{% cards %}} forms. */}} {{ if or (in $content "{{< cards") (in $content "{{% cards") }} {{ .Page.Store.Set "hasManualCards" true }} {{ end }} {{/* Strip out front matter (--- ... ---) */}} {{ $clean := replaceRE `(?s)^---.*?---\s*` "" $content }} {{/* ─── Stage 3: FORM CONVERSION ────────────────────────────────────── */}} {{/* Convert percent-bracket shortcodes to angle-bracket so RenderString processes them. Tab/tabs are converted back below so their inner content gets markdown-rendered. */}} {{ $clean = replaceRE `\{\{%\s*` "{{< " $clean }} {{ $clean = replaceRE `\s*%\}\}` " >}}" $clean }} {{/* Convert inner tab back to percent-bracket so its code blocks get markdown-rendered. Leave outer tabs as angle-bracket so its block-level HTML is inserted after markdown parsing, not re-parsed. Re-parsing would let CommonMark treat the wrapping
as a type-6 HTML block and terminate any surrounding ordered/unordered list. */}} {{ $clean = replaceRE `\{\{<\s*(tab\s[^>]*)>}}` "{{% $1%}}" $clean }} {{ $clean = replaceRE `\{\{<\s*(/tab)\s*>}}` "{{% $1 %}}" $clean }} {{/* Convert steps back to percent-bracket. Hextra's steps shortcode just wraps `.Inner`; with angle-bracket form the inner H3 headings stay as raw markdown text instead of rendering as

. */}} {{ $clean = replaceRE `\{\{<\s*(steps)\s*>}}` "{{% $1 %}}" $clean }} {{ $clean = replaceRE `\{\{<\s*(/steps)\s*>}}` "{{% $1 %}}" $clean }} {{/* Convert `include` back to percent-bracket. Hextra's include shortcode calls .RenderShortcodes on the target page and returns its raw markdown. With angle-bracket form, that markdown is inserted AFTER the outer markdown pass, so backticks, links, and headings appear as literal text. The percent form lets the outer markdown engine re-process the inserted content. Hextra explicitly documents include as a percent-form-only shortcode for this reason. */}} {{ $clean = replaceRE `\{\{<\s*(include\s[^>]*)>}}` "{{% $1%}}" $clean }} {{/* ─── Stage 4: ARG INJECTION ──────────────────────────────────────── */}} {{/* Adjust any relative doc links if needed */}} {{ $clean = replaceRE `\((\/?)docs\/` "(/" $clean }} {{/* Replace link and reuse shortcodes to pass version */}} {{ $clean = replaceRE `{{<\s*link-hextra\s*(.*?)>}}` (printf "{{< link-hextra $1 version=\"%s\" product=\"%s\" >}}" $version $product ) $clean }} {{ $clean = replaceRE `{{<\s*reuse\s*"([^"]+)"\s*>}}` (printf "{{< reuse \"%s\" \"%s\" \"%s\" >}}" "$1" $version $product ) $clean }} {{/* ─── Stage 5: VERSION REMAP ──────────────────────────────────────── */}} {{/* keepVersion: preserve version string, inject page version, convert to {{% %}} */}} {{ $clean = replaceRE `\{\{<\s*(version\s+(?:include|exclude)-if="[^"]+?")\s+keepVersion="true"\s*>}}` (printf `{{%% $1 version="%s" %%}}` $version) $clean }} {{/* Two-pass OSS→enterprise version remapping from [[params.versions]]. The percent form ({{% %}}) would trigger Hugo's parent-context pre-render of .Inner BEFORE version.html runs, and indented inner content would get wrapped in 
 by
     CommonMark's "4 spaces = code" rule before we ever see it. */}}

{{/* Table-row exception: a single-line version block whose inner content
     is a pipe-delimited row needs the percent form so the shortcode output
     re-enters the markdown stream and the pipes get parsed as cell
     delimiters by Goldmark. Angle form would emit opaque HTML inline and
     Goldmark would treat the whole pipe-string as one . The pattern
     match requires the opening and closing tags on the same line as the
     content, with the content starting and ending in `|` — narrow enough
     that it doesn't trip the indented-list or fence-adjacent cases that
     rely on staying in angle form. */}}
{{ $clean = replaceRE `\{\{<\s*(version\s+(?:include|exclude)-if="[^"]+"[^>]*?)\s*>}}(\s*\|[^\n]*?\|\s*)\{\{<\s*/version\s*>}}` "{{% $1 %}}$2{{% /version %}}" $clean }}

{{/* List-item exception: a version block whose inner content starts with
     a bullet or numbered list marker at column 0 needs the percent form
     so the shortcode output re-enters the markdown stream as a real list
     item. Angle form would emit the list marker as literal text inside
     the surrounding placeholder context — when the version wraps a single
     bullet adjacent to other bullets, Goldmark lazy-continues the prior
     `
  • ` and the `* [link](url)` survives as raw markdown (the `\[link]`
     leak in `merging.md`, `bedrock.md`). Matches both same-line wraps
     (`{{< version >}}* item{{< /version >}}`) and newline-prefixed multi-
     item blocks (`{{< version >}}\n* a\n* b\n{{< /version >}}`). Content
     must start at column 0 (no leading spaces before the marker) so we
     don't trip deeply-indented version blocks whose .Inner the upstream
     comment above protects from percent-form's 4-space-code-block rule. */}}
{{ $clean = replaceRE `\{\{<\s*(version\s+(?:include|exclude)-if="[^"]+"[^>]*?)\s*>}}((?:\r?\n)?(?:[*+\-]|[0-9]+\.)[ \t][\s\S]*?)\{\{<\s*/version\s*>}}` "{{% $1 %}}$2{{% /version %}}" $clean }}
