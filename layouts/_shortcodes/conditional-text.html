{{- /* conditional-text — include or exclude .Inner based on the page's build condition (resolved via utils/page-context for both multi-product-hub and single-site URL conventions). Hugo doesn't expose whether a shortcode was called in percent form or angle form, so this dispatches on the SHAPE of .Inner to pick an emit strategy. This is a faithful port of the docs-hub local override (the proven, leak-free behavior) — an earlier centralization pass dropped the trailing-step / table-row / dedent paths below, which regressed the docs hub's heavy "gme vs gmg" conditional content (conditional-text blocks that span a numbered-list step boundary rendered through reuse, breaking the surrounding list and leaking the following fenced code block as raw ```text```). Strategies: - TRAILING-STEP (numbered-list marker at column 0-3 of .Inner): the block continues a parent ordered list. Emit .Inner raw so Hugo re-flows it through the surrounding markdown stream. RenderString would produce a self-contained block that breaks the parent
    . - TABLE-ROW (FIRST non-blank line starts with `|`): the block IS a table-row fragment spliced into a surrounding table. Emit `.Inner` directly as template.HTML (same as the trailing-step path) so the row re-enters the markdown stream and merges after the `|---|` separator. Do NOT stringify it via printf/strings.*: that coerces the shortcode output (with its nested reuse/shortcode placeholders) to a plain string, and a nested {{< reuse >}} in the cell then has its rendered inline HTML (, ) escaped to <code> (the api-key / applyToRoutes "applyToDestinations" cell leak). No leading newline is needed (the call sits on its own line after the separator); adding one, or piping through safeHTML, breaks the merge and the row falls out as a literal `

    | … |`. - NO-MARKDOWN body (plain text / yaml / shell, e.g. inside a fenced code block): emit raw so RenderString doesn't mangle it (`#`→heading, "true"→smart quotes, --→en-dash) and escape it inside the fence. - EVERYTHING ELSE (inline/block markdown): dedent + RenderString(inline) + flatten, preserving a leading/trailing space for mid-sentence use. Shares the no-markdown heuristic (utils/has-markdown) and condition resolution (utils/page-context) with the version shortcode so the two can't drift. Does NOT route through utils/emit-inner: conditional-text has no display:"block" path and its raw-emit set differs from version's. */ -}} {{- $ctx := partial "utils/page-context.html" . -}} {{- $condition := $ctx.condition -}} {{- /* Trailing-step: .Inner continues a parent ordered list. */ -}} {{- $isTrailingStep := false -}} {{- if findRE `(?m)^[ \t]{0,3}[0-9]+\.\s+` .Inner 1 -}} {{- $isTrailingStep = true -}} {{- end -}} {{- /* Table-row: a body whose FIRST non-blank line is a `|`-delimited row, i.e. the block IS a table-row fragment (a conditional cell/row spliced into a surrounding table). Raw emit (with one leading newline) preserves the line structure so Goldmark still recognizes the `|---|---|` separator and parses the rows into the table. Must anchor on the FIRST non-blank line, not "any line contains a pipe": a numbered-list block whose steps happen to contain a markdown table (e.g. a "Setting | Description" review table inside step 2) is a LIST, not a table row. Letting isTableRow win there routes the whole list through the printf table-row path, and a later {{< reuse >}} in that block then has its rendered inline HTML (, ) escaped to <code> — the retry-timeout / mirror / rate-limit leak. Such a block is a trailing step and must raw-emit the whole body via {{ .Inner }} instead. */ -}} {{- $firstNonBlank := "" -}} {{- range (split .Inner "\n") -}} {{- if and (eq $firstNonBlank "") (ne (trim . " \t") "") -}} {{- $firstNonBlank = . -}} {{- end -}} {{- end -}} {{- $isTableRow := false -}} {{- if findRE `^[ \t]{0,3}\|` $firstNonBlank 1 -}} {{- $isTableRow = true -}} {{- end -}} {{- /* Fenced-code-block: a body whose FIRST non-blank line opens a fenced code block (``` or ~~~, indented 0-3 spaces). Raw-emit it (like the trailing-step path) so the fence re-enters the surrounding markdown stream and is rendered exactly ONCE by the page/reuse RenderString. The RenderString-then-flatten path (the else branch) renders the fence to a `

    …
    ` block HERE, and when that
       HTML lands inside a reused list item it's re-parsed as a CommonMark
       HTML block — the parent closes the `
  1. `/`
      ` and re-wraps the code
       guts in a `
      `, fragmenting the code block (the ext-auth-rate-limit
       deployer "To change the defaults" / "Disable the deployer" steps, and
       any `{{%% conditional-text %%}}`-gated fence inside a numbered step).
       Raw emit keeps the fence as markdown, so the single outer pass builds
       a well-formed `
    1. …
      2. `. Indentation is preserved (dedent is
       skipped below) so a 3-space-indented fence stays inside its list item.
       Anchors on the FIRST line so prose that merely mentions a fence later
       still takes the RenderString path. */ -}}
{{- $isFencedBlock := false -}}
{{- if findRE "^(```|~~~|\\{\\{[<%][^}]*prism)" (strings.TrimLeft " \t" $firstNonBlank) 1 -}}
  {{- $isFencedBlock = true -}}
{{- end -}}

{{- /* Full-table: a table-row body that ALSO contains a `|---|---|` delimiter
       line is a SELF-CONTAINED table (header + delimiter + rows), not a bare
       row fragment spliced into a table declared outside the block. The two
       need opposite emit strategies:

       - A row fragment must raw-emit (below) so it joins the surrounding
         table's markdown stream.
       - A self-contained table must be rendered to HTML here and emitted via
         safeHTML. Raw-emitting it breaks in two ways: angle form ({{< >}})
         bypasses markdown entirely, so the raw table never renders and leaks
         as literal `| … |` text; and percent form ({{% %}}) nested inside
         another RenderString (a reuse'd conref) reparses the raw table in a
         context that escapes inline HTML in cells, so `
      • ` cells leak as
         `<ul>`. Rendering here with display:block honors unsafe=true and
         flatten-rendered keeps it on one logical line, so the table always
         renders and cell HTML stays intact. Guards solo-io/docs about-databases
         #byo-local and the angle-form portal/about table. */ -}}
{{- $isFullTable := false -}}
{{- if and $isTableRow (findRE `(?m)^[ \t]{0,3}\|?[ \t]*:?-+:?[ \t]*(\|[ \t]*:?-+:?[ \t]*)+\|?[ \t]*$` .Inner 1) -}}
  {{- $isFullTable = true -}}
{{- end -}}

{{- /* Dedent .Inner for the RenderString path so block content at deep
       list-item indentation doesn't trip CommonMark's "4 spaces = code
       block" rule. Skipped for the trailing-step raw path. */ -}}
{{- $inner := .Inner -}}
{{- if not (or $isTrailingStep $isFencedBlock) -}}
  {{- $lines := split $inner "\n" -}}
  {{- $minIndent := -1 -}}
  {{- range $lines -}}
    {{- $trimmed := strings.TrimLeft " \t" . -}}
    {{- if ne $trimmed "" -}}
      {{- $thisIndent := sub (len .) (len $trimmed) -}}
      {{- if or (eq $minIndent -1) (lt $thisIndent $minIndent) -}}
        {{- $minIndent = $thisIndent -}}
      {{- end -}}
    {{- end -}}
  {{- end -}}
  {{- if gt $minIndent 0 -}}
    {{- $dedented := slice -}}
    {{- range $lines -}}
      {{- if ge (len .) $minIndent -}}
        {{- $dedented = $dedented | append (substr . $minIndent) -}}
      {{- else -}}
        {{- $dedented = $dedented | append . -}}
      {{- end -}}
    {{- end -}}
    {{- $inner = delimit $dedented "\n" -}}
  {{- end -}}
{{- end -}}

{{- /* No-markdown heuristic on the dedented $inner (shared with version). */ -}}
{{- $hasMarkdown := partial "utils/has-markdown.html" $inner -}}

{{- if ne $condition "" -}}
  {{- $include := .Get "include-if" | default "" -}}
  {{- $include_if := split $include "," -}}
  {{- $exclude := .Get "exclude-if" -}}
  {{- $exclude_if := split $exclude "," -}}

  {{- if and (in $include_if $condition) (in $exclude_if $condition) -}}
    {{- errorf "Build condition %q appears in both include-if and exclude-if parameters of conditional-text shortcode on page %s" $condition .Position -}}
  {{- end -}}

  {{- if isset $.Params "include-if" -}}
    {{- if in $include_if $condition -}}
      {{- if $isFullTable -}}
        {{- /* Self-contained table: render to HTML (display:block keeps cell
               HTML, since unsafe=true) and flatten so angle form emits it
               directly and percent form re-renders it as a single HTML block
               instead of re-escaping the cells. */ -}}
        {{- $rendered_html := $.Page.RenderString (dict "display" "block") $inner -}}
        {{- $rendered_html = partial "utils/flatten-rendered.html" $rendered_html -}}
        {{- $rendered_html | safeHTML -}}
      {{- else if $isTableRow -}}
{{ $.Inner }}
      {{- else if or $isTrailingStep $isFencedBlock (not $hasMarkdown) -}}
{{ $.Inner }}
      {{- else -}}
        {{- $rendered_html := $.Page.RenderString (dict "display" "inline") $inner -}}
        {{- $rendered_html = partial "utils/flatten-rendered.html" $rendered_html -}}
        {{- if hasPrefix $.Inner " " }}{{ " " | safeHTML }}{{ end -}}
        {{- $rendered_html | safeHTML -}}
        {{- if hasSuffix $.Inner " " }}{{ " " | safeHTML }}{{ end -}}
      {{- end -}}
    {{- end -}}
  {{- else if isset $.Params "exclude-if" -}}
    {{- if not (in $exclude_if $condition) -}}
      {{- if $isFullTable -}}
        {{- /* Self-contained table: render to HTML (display:block keeps cell
               HTML, since unsafe=true) and flatten so angle form emits it
               directly and percent form re-renders it as a single HTML block
               instead of re-escaping the cells. */ -}}
        {{- $rendered_html := $.Page.RenderString (dict "display" "block") $inner -}}
        {{- $rendered_html = partial "utils/flatten-rendered.html" $rendered_html -}}
        {{- $rendered_html | safeHTML -}}
      {{- else if $isTableRow -}}
{{ $.Inner }}
      {{- else if or $isTrailingStep $isFencedBlock (not $hasMarkdown) -}}
{{ $.Inner }}
      {{- else -}}
        {{- $rendered_html := $.Page.RenderString (dict "display" "inline") $inner -}}
        {{- $rendered_html = partial "utils/flatten-rendered.html" $rendered_html -}}
        {{- if hasPrefix $.Inner " " }}{{ " " | safeHTML }}{{ end -}}
        {{- $rendered_html | safeHTML -}}
        {{- if hasSuffix $.Inner " " }}{{ " " | safeHTML }}{{ end -}}
      {{- end -}}
    {{- end -}}
  {{- end -}}
{{- end -}}
