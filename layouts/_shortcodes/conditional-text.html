{{- /* Use the shared page-context partial so this shortcode works for both URL conventions: multi-product hubs (condition from Site.Params.buildCondition) and single-site repos that encode the section in the URL (condition derived from the URL section). */ -}} {{- $ctx := partial "utils/page-context.html" . -}} {{- $condition := $ctx.condition -}} {{- /* No-markdown heuristic, mirrored from version.html. If .Inner has no inline-formatting or list markers, RenderString would only damage it (turning `#` into

, "foo" into curly quotes, -- into –, plain text into HTML-escaped angle text when the shortcode body lands inside a fenced code block). Emit raw in that case. The HTML-tag regex requires a lowercase tag name terminated by space/`/`/`>` so yaml placeholders like and don't trigger a false positive. */ -}} {{- $hasMarkdown := false -}} {{- if findRE `\*\*` .Inner 1 -}}{{- $hasMarkdown = true -}}{{- end -}} {{- if not $hasMarkdown -}} {{- if findRE "`" .Inner 1 -}}{{- $hasMarkdown = true -}}{{- end -}} {{- end -}} {{- if not $hasMarkdown -}} {{- if findRE `(?m)^[ \t]{0,3}([*+\-][ \t]|[0-9]+\.[ \t])` .Inner 1 -}}{{- $hasMarkdown = true -}}{{- end -}} {{- end -}} {{- if not $hasMarkdown -}} {{- if findRE `\[[^\]]*\]\(` .Inner 1 -}}{{- $hasMarkdown = true -}}{{- end -}} {{- end -}} {{- if ne $condition "" -}} {{- $include := .Get "include-if" | default "" -}} {{- $include_if := split $include "," -}} {{- $exclude := .Get "exclude-if" -}} {{- $exclude_if := split $exclude "," -}} {{- if and (in $include_if $condition) (in $exclude_if $condition) -}} {{- errorf "Build condition %q appears in both include-if and exclude-if parameters of conditional-txt shortcode on page %s" $condition .Position -}} {{- end -}} {{- if isset $.Params "include-if" -}} {{- if in $include_if $condition -}} {{- if not $hasMarkdown -}} {{ $.Inner }} {{- else -}} {{- $rendered := $.Page.RenderString (dict "display" "inline") $.Inner -}} {{- if hasPrefix $.Inner " " }}{{ " " | safeHTML }}{{ end -}} {{- $rendered | safeHTML -}} {{- if hasSuffix $.Inner " " }}{{ " " | safeHTML }}{{ end -}} {{- end -}} {{- else -}} {{- end -}} {{- else -}} {{- if isset $.Params "exclude-if" -}} {{- if in $exclude_if $condition -}} {{- else -}} {{- if not $hasMarkdown -}} {{ $.Inner }} {{- else -}} {{- $rendered := $.Page.RenderString (dict "display" "inline") $.Inner -}} {{- if hasPrefix $.Inner " " }}{{ " " | safeHTML }}{{ end -}} {{- $rendered | safeHTML -}} {{- if hasSuffix $.Inner " " }}{{ " " | safeHTML }}{{ end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}} {{- end -}}
