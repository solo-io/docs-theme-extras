{{- /* Shortcode: card Renders a styled card link with an optional icon or image, title, and subtitle. Emits the same .section-card markup the parent-section auto-card grid uses, so manual {{< card >}} blocks look identical to auto-generated child cards. Parameters: - link: (optional) A direct URL for the card link (used as-is). - path: (optional) A section-relative path resolved to the current docs section (e.g., "/tutorials/basic/"). - title: The card title (aliased as "header"). - subtitle: (optional) Short description below the title (aliased as "description"); supports Markdown. - icon: (optional) Name of an icon to render via utils/icon.html. - image: (optional) Image URL to display at the top of the card. - alt: (optional) Alt text for the image; defaults to the title. Title and subtitle are emitted as with display: block via CSS — NOT

— so the card stays valid HTML when it's wrapped by markdownify (e.g., from a percent-form {{%% %%}} shortcode). A nested

inside the outer

Goldmark adds would trigger the browser's auto-close-the-outer-

behavior, closing the too early and orphaning the description. Example: {{< card title="Get started" subtitle="Learn the basics" path="/getting-started/" icon="rocket" >}} */ -}} {{- $link := .Get "link" -}} {{- /* If the link argument contains a nested shortcode call (passed via backtick quoting, e.g. link=`{{< link path="foo" >}}`), Hugo does not automatically expand it — backtick-quoted args are raw strings. Detect the pattern and evaluate it through RenderString so the inner shortcode resolves to a URL before it reaches the href attribute. */ -}} {{- if strings.Contains $link "{{<" -}} {{- $link = $.Page.RenderString $link | plainify | strings.TrimSpace -}} {{- end -}} {{- $path := .Get "path" -}} {{- if $path }} {{- $pathTrimmed := strings.TrimPrefix "/" $path -}} {{- /* Resolve the docs section prefix via the shared page-context partial so this shortcode picks up every section the partial knows about (kubernetes, standalone, envoy, agentgateway, ...) without hardcoding a regex here. Falls back to FirstSection.RelPermalink for pages outside the /docs/

// shape. */ -}} {{- $ctx := partial "utils/page-context.html" . -}} {{- $section := $ctx.prefix -}} {{- if not $section }}{{ $section = strings.TrimSuffix "/" .Page.FirstSection.RelPermalink }}{{ end -}} {{- $section = replaceRE `^https?://[^/]+` "" $section -}} {{- $section = replaceRE `^/localhost:[^/]+` "" $section -}} {{- if not (hasPrefix $section "/") }}{{ $section = printf "/%s" $section }}{{ end -}} {{- $link = printf "%s/%s" $section $pathTrimmed -}} {{- if and (not (strings.HasSuffix $link "/")) (not (in $link "#")) }}{{ $link = printf "%s/" $link }}{{ end -}} {{- range (seq 10) }}{{ $link = replace $link "//" "/" }}{{ end -}} {{- end -}} {{- $title := .Get "title" | default (.Get "header") -}} {{- $subtitle := .Get "subtitle" | default (.Get "description") -}} {{- $icon := .Get "icon" -}} {{- $image := .Get "image" -}} {{- $imageSrc := $image -}} {{- if and $image (not (or (strings.HasPrefix $image "http") (strings.HasPrefix $image "/"))) -}} {{- /* Asset-relative path → resolve through Hugo's asset pipeline so files under assets/ are published (a bare "assets/img/x.png" would otherwise render as a page-relative URL and 404). resources.Get is relative to the assets/ dir, so strip a leading "assets/" if the author wrote it. */ -}} {{- with resources.Get (strings.TrimPrefix "assets/" $image) -}} {{- $imageSrc = .RelPermalink -}} {{- else -}} {{- warnf "card.html: image %q not found under assets/ (resources.Get returned nil); leaving src as-is" $image -}} {{- end -}} {{- end -}} {{- $alt := .Get "alt" | default $title -}} {{- if $image -}} {{- end -}} {{- if $icon -}} {{ partial "utils/icon.html" (dict "name" $icon "attributes" "class=\"section-card-icon\"") }} {{- end -}} {{- if $title }}{{ $title }}{{ end -}} {{- if $subtitle }}{{ $subtitle | markdownify }}{{ end -}}
