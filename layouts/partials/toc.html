{{/* Custom TOC that parses headings from rendered .Content HTML. This is needed because Hugo's .Fragments.Headings doesn't detect headings injected via rebase/reuse shortcodes. */}} {{- $toc := .Params.toc | default true -}} {{- $onThisPage := (T "onThisPage") | default "On this page" -}} {{- $backToTop := (T "backToTop") | default "Scroll to top" -}} {{- $startLevel := 2 -}} {{- $endLevel := 4 -}}
{{- if $toc }}
{{- /* Extract headings from rendered Content HTML. Hextra renders headings as:

Title......

The id is on a span inside the heading, not on the h tag itself. */ -}} {{- $headings := slice -}} {{- range findRE `(?is)]*>.*?` .Content -}} {{- $level := substr . 2 1 | int -}} {{- if and (ge $level $startLevel) (le $level $endLevel) -}} {{- $id := "" -}} {{- /* Try id on the heading tag itself */ -}} {{- if findRE `(?is)]+id="` . -}} {{- $id = replaceRE `(?is).*?id="([^"]*)".*` "$1" . -}} {{- /* Try id on a span inside the heading (hextra pattern) */ -}} {{- else if findRE `(?is)]+id="` . -}} {{- $id = replaceRE `(?is).*]*id="([^"]*)".*` "$1" . -}} {{- end -}} {{- $text := replaceRE `(?is)]*>(.*?)` "$1" . -}} {{- $text = $text | plainify | htmlUnescape -}} {{- $text = replaceRE `\s*link$` "" $text -}} {{- $text = trim $text " \t\n" -}} {{- if and $id $text -}} {{- $headings = $headings | append (dict "id" $id "level" $level "text" $text) -}} {{- end -}} {{- end -}} {{- end -}} {{- with $headings -}}
