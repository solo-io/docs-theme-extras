{{- /* inner-shape — shared shape detection for the version and conditional-text shortcodes. Hugo doesn't expose whether a shortcode was called as {{% %}} or {{< >}}, so both shortcodes dispatch on the SHAPE of .Inner to pick an emit strategy. This partial centralizes that detection so the regexes live in one place and can't drift between the two shortcodes. Input: the shortcode context (.) — only .Inner is read. Output: a dict with .isTrailingStep — first non-blank line is a list marker, OR last non-blank line is a bare orphan marker (raw-emit so the body re-flows into a parent list). .inner — .Inner, dedented for the non-trailing-step path (so deep list-item indentation doesn't trip CommonMark's 4-space code-block rule in RenderString). .hasMarkdown — has **, backticks, lowercase HTML tags, list markers, or [text](url) links. .isBlock — starts a heading / fenced code / table row at line start → needs display:"block" rendering. .isRowContent — every non-blank line is a pipe-delimited table row. */ -}} {{- $isTrailingStep := false -}} {{- $firstNonBlank := "" -}} {{- $lastNonBlank := "" -}} {{- range (split .Inner "\n") -}} {{- if ne (trim . " \t") "" -}} {{- if eq $firstNonBlank "" -}} {{- $firstNonBlank = . -}} {{- end -}} {{- $lastNonBlank = . -}} {{- end -}} {{- end -}} {{- /* Trailing-step detection — two independent triggers, matching the proven docs-hub override (which gates NEITHER on the presence of a fence/heading). 1. FIRST-LINE marker: the body opens on a `N.` / `-` list marker, so it continues a parent list and must raw-emit to re-flow into that list. 2. LAST-LINE bare-orphan marker: a trailing `3.` where the version closes on the next step's marker (the docs-hub extauth-opa-sidecar.md shape: para+bullets+fence handing off to the next list item). NOTE: an earlier centralization pass gated trigger 1 on a "has heading / fenced code" guard to fix the kgw session-persistence leak (an angle-form {{< version >}} wrapping a list step + yaml fence that raw-emit leaked). That guard REGRESSED the docs hub: a percent-form version block glued to a closing fence (e.g. ` ```{{% version %}} ` in jwt-2x-provider-header.md, and ~70 istio/gme pages) was forced to render-as-block, so its
    HTML glued straight after the ` ``` ` with no newline and the fence never closed — leaking ` ```
      ` into the code block. The guard was a stand-in for "angle vs percent", a distinction Hugo doesn't expose here; reuse.html/rebase.html already rewrite angle→percent for these shapes, so the guard is removed to restore the override's behavior. Both consumers (docs hub + kgw-oss) verified leak-free without it. */ -}} {{- if findRE `^[ \t]{0,3}([*+\-][ \t]|[0-9]+\.[ \t])` $firstNonBlank 1 -}} {{- $isTrailingStep = true -}} {{- end -}} {{- if findRE `^[ \t]{0,3}([*+\-]|[0-9]+\.)[ \t]*$` $lastNonBlank 1 -}} {{- $isTrailingStep = true -}} {{- end -}} {{- /* Fenced-code-block: the body OPENS on an INDENTED fenced code block (``` or ~~~, as a list-item continuation). Like trailing-step, this must raw-emit so the fence re-enters the surrounding markdown stream and is rendered exactly ONCE by the page/reuse RenderString. The RenderString-then-flatten path renders the fence to a `
      …
      ` block
       HERE; when that HTML lands inside a reused list item it's re-parsed as a
       CommonMark HTML block — the parent closes the `
    1. `/`
        ` and re-wraps
       the code guts in a `
        `, fragmenting the block (the kgw jwt-provider
       RBAC step, the gme/gmg opa-sidecar / opa-byo / mirror steps: a
       `{{%% version %%}}`-gated yaml fence inside a numbered step). Raw emit
       keeps the fence as markdown so the single outer pass builds a well-formed
       `
      1. …
        2. `. Distinct from the no-markdown path (a version block
       INSIDE a fence, whose body is bare yaml/shell): there the fence delimiters
       live OUTSIDE the block, so $firstNonBlank is yaml, not ```; here the
       block WRAPS the fence, so $firstNonBlank IS the ``` opener. Indentation is
       preserved (emit-inner raw-emits .rawInner, never the dedented copy).

       SAFETY (angle form): an angle-form `{{< version >}}` wrapping a bare fence
       would leak the literal ```` ```lang ```` as text if raw-emitted, since
       angle output bypasses markdown. reuse.html/rebase.html rewrite such
       angle blocks to percent form before this partial runs (the same
       angle→percent pass that already covers the list-marker shapes), so by the
       time we see a fence-leading body it is percent form (or authored directly
       on a page, where a single render needs no raw-emit and doesn't fragment). */ -}}
{{- /* Matches an INDENTED fence (``` / ~~~) OR an indented leading `prism`
       shortcode — a fenced/highlighted block sitting as a list-item
       continuation (the marker is OUTSIDE this block, on the preceding line),
       which fragments when block-rendered inside a reused list item. The
       leading-whitespace requirement (`[ \t]+`) is load-bearing: it scopes
       raw-emit to list-continuation fences and EXCLUDES a column-0 fence — a
       fence glued to the shortcode tag (` {{< version >}}```yaml `) or a
       top-level gated fence. Those must keep the RenderString path: raw-emitting
       a column-0 fence from an ANGLE block (not rewritten to percent because it
       isn't a list-marker shape) leaks the literal ```` ```lang ```` as text.
       Any list-continuation indent works (`1. `=3, `13. `=4 spaces); raw-emit
       preserves it via .rawInner. Distinct from the no-markdown path (a block
       INSIDE a fence, body = bare yaml/shell, fence delimiters outside). */ -}}
{{- $isFencedBlock := false -}}
{{- if findRE "^[ \t]+(```|~~~|\\{\\{[<%][^}]*prism)" $firstNonBlank 1 -}}
  {{- $isFencedBlock = true -}}
{{- end -}}

{{- $inner := .Inner -}}
{{- if not (or $isTrailingStep $isFencedBlock) -}}
  {{- $lines := split $inner "\n" -}}
  {{- $minIndent := -1 -}}
  {{- range $lines -}}
    {{- $trimmed := strings.TrimLeft " \t" . -}}
    {{- if ne $trimmed "" -}}
      {{- $thisIndent := sub (len .) (len $trimmed) -}}
      {{- if or (eq $minIndent -1) (lt $thisIndent $minIndent) -}}
        {{- $minIndent = $thisIndent -}}
      {{- end -}}
    {{- end -}}
  {{- end -}}
  {{- if gt $minIndent 0 -}}
    {{- $dedented := slice -}}
    {{- range $lines -}}
      {{- if ge (len .) $minIndent -}}
        {{- $dedented = $dedented | append (substr . $minIndent) -}}
      {{- else -}}
        {{- $dedented = $dedented | append . -}}
      {{- end -}}
    {{- end -}}
    {{- $inner = delimit $dedented "\n" -}}
  {{- end -}}
{{- end -}}

{{- /* hasMarkdown via the shared utils/has-markdown partial (same regexes the
       conditional-text shortcode uses) so the heuristic can't drift. Computed
       on the dedented $inner, matching the pre-refactor version.html. */ -}}
{{- $hasMarkdown := partial "utils/has-markdown.html" $inner -}}

{{- $isBlock := false -}}
{{- if findRE "(?m)^[ \\t]{0,3}(#{1,6}\\s|`{3}|\\|[^\\n]*\\|)" $inner 1 -}}
  {{- $isBlock = true -}}
{{- end -}}

{{- $isRowContent := false -}}
{{- $hasAnyRow := false -}}
{{- $allLinesAreRows := true -}}
{{- range (split $inner "\n") -}}
  {{- $t := trim . " \t" -}}
  {{- if ne $t "" -}}
    {{- if and (hasPrefix $t "|") (hasSuffix $t "|") -}}
      {{- $hasAnyRow = true -}}
    {{- else -}}
      {{- $allLinesAreRows = false -}}
    {{- end -}}
  {{- end -}}
{{- end -}}
{{- if and $hasAnyRow $allLinesAreRows -}}
  {{- $isRowContent = true -}}
{{- end -}}

{{- return (dict "isTrailingStep" $isTrailingStep "isFencedBlock" $isFencedBlock "inner" $inner "hasMarkdown" $hasMarkdown "isBlock" $isBlock "isRowContent" $isRowContent) -}}
