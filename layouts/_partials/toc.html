{{/* Right-hand Table of Contents. Regex-extracts headings from .Content (h2-h4 by default) and renders them as a nested list. Centralized from kgateway/agentgateway-oss — consumers no longer need a local toc.html. Styling lives in docs-theme-extras.css under .solo-toc-* class names plus the existing .hextra-toc rules. No inline styles or Tailwind utility classes embedded in this template, so it isn't subject to bundle-coverage quirks. */}} {{- $toc := .Params.toc | default true -}} {{- $onThisPage := (T "onThisPage") | default "On this page" -}} {{- $backToTop := (T "backToTop") | default "Scroll to top" -}} {{- $startLevel := 2 }} {{- $endLevel := 4 }} {{- $minNumHeadings := 2 }} {{- /* Collect headings with regex */ -}} {{- $headings := slice }} {{- range findRE `(?is)` .Content }} {{- $level := substr . 2 1 | int }} {{- if and (ge $level $startLevel) (le $level $endLevel) }} {{- $text := replaceRE `(?is)(.+?)` "$1" . | plainify | htmlUnescape }} {{- $id := "" }} {{- if findRE `\s+id=` . }} {{- $id = replaceRE `(?is).+?\s+id=(?:\x22|\x27)?(.*?)(?:\x22|\x27)?[\s>].+` "$1" . }} {{- end }} {{- $headings = $headings | append (dict "id" $id "level" $level "text" $text) }} {{- end }} {{- end }} {{/* ---------- recursive renderer (consumes list so no duplicates) ---------- */}} {{- define "render-list" -}}
    {{- template "render-items" . -}}
{{- end }} {{- define "render-items" -}} {{- $headings := .headings -}} {{- $parentLevel := .parentLevel -}} {{- $n := len $headings -}} {{- if gt $n 0 -}} {{- $h := index $headings 0 -}}
  • {{ $h.text }} {{- /* collect immediate children that come after this heading */ -}} {{- $subs := slice -}} {{- if gt $n 1 -}} {{- range $j := seq 1 (sub $n 1) -}} {{- $nh := index $headings $j -}} {{- if le $nh.level $h.level -}} {{- break -}} {{- end -}} {{- $subs = $subs | append $nh -}} {{- end -}} {{- end -}} {{- if gt (len $subs) 0 -}} {{- template "render-list" (dict "headings" $subs "parentLevel" $h.level) -}} {{- end -}}
    • {{- /* render the rest of headings at this level */ -}} {{- $consumed := add 1 (len $subs) -}} {{- if lt $consumed $n -}} {{- $rest := slice -}} {{- range $k := seq $consumed (sub $n 1) -}} {{- $rest = $rest | append (index $headings $k) -}} {{- end -}} {{- template "render-items" (dict "headings" $rest "parentLevel" $parentLevel) -}} {{- end -}} {{- end -}} {{- end }} {{/* ---------- TOC container ---------- */}} {{- if and $toc (ge (len $headings) $minNumHeadings) }}

    {{ $onThisPage }}

    {{ template "render-list" (dict "headings" $headings "parentLevel" $startLevel) }}
    {{- end }}
